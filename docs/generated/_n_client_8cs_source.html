183  public void Send<T>(INMessage<T> message, Action<T> callback, Action<INError> errback)
184  {
185  // Set a collation ID to dispatch callbacks on receive
186  string collationId = Guid.NewGuid().ToString();
187  message.SetCollationId(collationId);
188 
189  // Track callbacks for message
190  var pair = new KeyValuePair<Action<object>, Action<INError>>((data) =>
191  {
192  callback((T)data);
193  }, errback);
194  collationIds.Add(collationId, pair);
195 
196  var stream = new MemoryStream();
197  message.Payload.WriteTo(stream);
198  Logger.TraceFormatIf(Trace, "SocketWrite: {0}", message.Payload);
199  transport.SendAsync(stream.ToArray(), (bool completed) =>
200  {
201  if (!completed)
202  {
203  // The write may have failed; don't track it
204  collationIds.Remove(collationId);
205  }
206  });
207  }

215  private void authenticate(string path,
216  AuthenticateRequest payload,
217  string langHeader,
218  Action<INSession> callback,
219  Action<INError> errback)
220  {
221  // Add a collation ID for logs
222  payload.CollationId = Guid.NewGuid().ToString();
223 
224  var scheme = (SSL) ? "https" : "http";
225  var uri = new UriBuilder(scheme, Host, unchecked((int)Port), path).Uri;
226  Logger.TraceFormatIf(Trace, "Url={0}, Payload={1}", uri, payload);
227 
228  byte[] buffer = Encoding.UTF8.GetBytes(ServerKey + ":");
229  var authHeader = String.Concat("Basic ", Convert.ToBase64String(buffer));
230 
231  TimeSpan span = DateTime.UtcNow - new DateTime(1970, 1, 1, 0, 0, 0, DateTimeKind.Utc);
232  transport.Post(uri.ToString(), payload, authHeader, langHeader, Timeout, ConnectTimeout, (data) =>
233  {
234  AuthenticateResponse authResponse = AuthenticateResponse.Parser.ParseFrom(data);
235  Logger.TraceFormatIf(Trace, "DecodedResponse={0}", authResponse);
236 
237  switch (authResponse.PayloadCase)
238  {
239  case AuthenticateResponse.PayloadOneofCase.Session:
240  callback(new NSession(authResponse.Session.Token, System.Convert.ToInt64(span.TotalMilliseconds)));
241  break;
242  case AuthenticateResponse.PayloadOneofCase.Error:
243  errback(new NError(authResponse.Error));
244  break;
245  case AuthenticateResponse.PayloadOneofCase.None:
246  Logger.Error("Received invalid response from server");
247  break;
248  default:
249  Logger.Error("Received invalid response from server");
250  break;
251  }
252  }, (e) =>
253  {
254  errback(new NError(e.Message));
255  });
256  }

267  private void onMessage(Envelope message)
268  {
269  // Handle realtime messages
270  switch (message.PayloadCase)
271  {
272  case Envelope.PayloadOneofCase.Heartbeat:
273  ServerTime = message.Heartbeat.Timestamp;
274  return;
275  case Envelope.PayloadOneofCase.MatchData:
276  if (OnMatchData != null)
277  {
278  OnMatchData(this, new NMatchDataEventArgs(new NMatchData(message.MatchData)));
279  }
280  return;
281  case Envelope.PayloadOneofCase.MatchPresence:
282  if (OnMatchPresence != null)
283  {
284  OnMatchPresence(this, new NMatchPresenceEventArgs(new NMatchPresence(message.MatchPresence)));
285  }
286  return;
287  case Envelope.PayloadOneofCase.TopicMessage:
288  if (OnTopicMessage != null)
289  {
290  OnTopicMessage(this, new NTopicMessageEventArgs(new NTopicMessage(message.TopicMessage)));
291  }
292  return;
293  case Envelope.PayloadOneofCase.TopicPresence:
294  if (OnTopicPresence != null)
295  {
296  OnTopicPresence(this, new NTopicPresenceEventArgs(new NTopicPresence(message.TopicPresence)));
297  }
298  return;
299  }
300 
301  var collationId = message.CollationId;
302  var pair = collationIds[collationId];
303  collationIds.Remove(collationId);
304 
305  switch (message.PayloadCase)
306  {
307  case Envelope.PayloadOneofCase.None:
308  pair.Key(true);
309  break;
310  case Envelope.PayloadOneofCase.Error:
311  var error = new NError(message.Error.Reason);
312  if (collationId != null)
313  {
314  pair.Value(error);
315  }
316  else
317  {
318  if (OnError != null)
319  {
320  OnError(this, new NErrorEventArgs(error));
321  }
322  }
323  break;
324  case Envelope.PayloadOneofCase.Friends:
325  var friends = new List<INFriend>();
326  foreach (var friend in message.Friends.Friends)
327  {
328  friends.Add(new NFriend(friend));
329  }
330  pair.Key(new NResultSet<INFriend>(friends, null));
331  break;
332  case Envelope.PayloadOneofCase.Group:
333  pair.Key(new NGroup(message.Group.Group));
334  break;
335  case Envelope.PayloadOneofCase.GroupUsers:
336  var groupUsers = new List<INGroupUser>();
337  foreach (var groupUser in message.GroupUsers.Users)
338  {
339  groupUsers.Add(new NGroupUser(groupUser));
340  }
341  pair.Key(new NResultSet<INGroupUser>(groupUsers, null));
342  break;
343  case Envelope.PayloadOneofCase.Groups:
344  var groups = new List<INGroup>();
345  foreach (var group in message.Groups.Groups)
346  {
347  groups.Add(new NGroup(group));
348  }
349  pair.Key(new NResultSet<INGroup>(groups, new NCursor(message.Groups.Cursor.ToByteArray())));
350  break;
351  case Envelope.PayloadOneofCase.Match:
352  pair.Key(new NMatch(message.Match));
353  break;
354  case Envelope.PayloadOneofCase.Self:
355  pair.Key(new NSelf(message.Self.Self));
356  break;
357  case Envelope.PayloadOneofCase.StorageKey:
358  var keys = new List<INStorageKey>();
359  foreach (var key in message.StorageKey.Keys)
360  {
361  keys.Add(new NStorageKey(key));
362  }
363  pair.Key(new NResultSet<INStorageKey>(keys, null));
364  break;
365  case Envelope.PayloadOneofCase.StorageData:
366  var storageData = new List<INStorageData>();
367  foreach (var data in message.StorageData.Data)
368  {
369  storageData.Add(new NStorageData(data));
370  }
371  pair.Key(new NResultSet<INStorageData>(storageData, null));
372  break;
373  case Envelope.PayloadOneofCase.Topic:
374  pair.Key(new NTopic(message.Topic));
375  break;
376  case Envelope.PayloadOneofCase.TopicMessageAck:
377  pair.Key(new NTopicMessageAck(message.TopicMessageAck));
378  break;
379  case Envelope.PayloadOneofCase.TopicMessages:
380  var topicMessages = new List<INTopicMessage>();
381  foreach (var topicMessage in message.TopicMessages.Messages)
382  {
383  topicMessages.Add(new NTopicMessage(topicMessage));
384  }
385  pair.Key(new NResultSet<INTopicMessage>(topicMessages, new NCursor(message.TopicMessages.Cursor.ToByteArray())));
386  break;
387  case Envelope.PayloadOneofCase.Users:
388  var users = new List<INUser>();
389  foreach (var user in message.Users.Users)
390  {
391  users.Add(new NUser(user));
392  }
393  pair.Key(new NResultSet<INUser>(users, null));
394  break;
395  default:
396  Logger.TraceFormatIf(Trace, "Unrecognized message: {0}", message);
397  break;
398  }
