329  private void onMessage(Envelope message)
330  {
331  // Handle realtime messages
332  switch (message.PayloadCase)
333  {
334  case Envelope.PayloadOneofCase.Heartbeat:
335  ServerTime = message.Heartbeat.Timestamp;
336  return;
337  case Envelope.PayloadOneofCase.MatchmakeMatched:
338  if (OnMatchmakeMatched != null)
339  {
340  OnMatchmakeMatched(new NMatchmakeMatched(message.MatchmakeMatched));
341  }
342  return;
343  case Envelope.PayloadOneofCase.MatchData:
344  if (OnMatchData != null)
345  {
346  OnMatchData(new NMatchData(message.MatchData));
347  }
348  return;
349  case Envelope.PayloadOneofCase.MatchPresence:
350  if (OnMatchPresence != null)
351  {
352  OnMatchPresence(new NMatchPresence(message.MatchPresence));
353  }
354  return;
355  case Envelope.PayloadOneofCase.TopicMessage:
356  if (OnTopicMessage != null)
357  {
358  OnTopicMessage(new NTopicMessage(message.TopicMessage));
359  }
360  return;
361  case Envelope.PayloadOneofCase.TopicPresence:
362  if (OnTopicPresence != null)
363  {
364  OnTopicPresence(new NTopicPresence(message.TopicPresence));
365  }
366  return;
367  case Envelope.PayloadOneofCase.LiveNotifications:
368  if (OnNotification != null)
369  {
370  foreach (var n in message.LiveNotifications.Notifications_)
371  {
372  OnNotification(new NNotification(n));
373  }
374  }
375  return;
376  }
377 
378  var collationId = message.CollationId;
379  var pair = collationIds[collationId];
380  collationIds.Remove(collationId);
381 
382  switch (message.PayloadCase)
383  {
384  case Envelope.PayloadOneofCase.None:
385  pair.Key(true);
386  break;
387  case Envelope.PayloadOneofCase.Error:
388  var error = new NError(message.Error, collationId);
389  if (collationId != null)
390  {
391  pair.Value(error);
392  }
393  else
394  {
395  if (OnError != null)
396  {
397  OnError(error);
398  }
399  }
400  break;
401  case Envelope.PayloadOneofCase.Friends:
402  var friends = new List<INFriend>();
403  foreach (var friend in message.Friends.Friends)
404  {
405  friends.Add(new NFriend(friend));
406  }
407  pair.Key(new NResultSet<INFriend>(friends, null));
408  break;
409  case Envelope.PayloadOneofCase.GroupUsers:
410  var groupUsers = new List<INGroupUser>();
411  foreach (var groupUser in message.GroupUsers.Users)
412  {
413  groupUsers.Add(new NGroupUser(groupUser));
414  }
415  pair.Key(new NResultSet<INGroupUser>(groupUsers, null));
416  break;
417  case Envelope.PayloadOneofCase.Groups:
418  var groups = new List<INGroup>();
419  foreach (var group in message.Groups.Groups)
420  {
421  groups.Add(new NGroup(group));
422  }
423  pair.Key(new NResultSet<INGroup>(groups, new NCursor(message.Groups.Cursor)));
424  break;
425  case Envelope.PayloadOneofCase.GroupsSelf:
426  var groupsSelf = new List<INGroupSelf>();
427  foreach (var gs in message.GroupsSelf.GroupsSelf)
428  {
429  groupsSelf.Add(new NGroupSelf(gs));
430  }
431  pair.Key(new NResultSet<INGroupSelf>(groupsSelf, null));
432  break;
433  case Envelope.PayloadOneofCase.MatchmakeTicket:
434  pair.Key(new NMatchmakeTicket(message.MatchmakeTicket));
435  break;
436  case Envelope.PayloadOneofCase.Match:
437  pair.Key(new NMatch(message.Match));
438  break;
439  case Envelope.PayloadOneofCase.Matches:
440  var matches = new List<INMatch>();
441  foreach (var m in message.Matches.Matches)
442  {
443  matches.Add(new NMatch(m));
444  }
445  pair.Key(new NResultSet<INMatch>(matches, null));
446  break;
447  case Envelope.PayloadOneofCase.Self:
448  pair.Key(new NSelf(message.Self.Self));
449  break;
450  case Envelope.PayloadOneofCase.StorageKeys:
451  var keys = new List<INStorageKey>();
452  foreach (var key in message.StorageKeys.Keys)
453  {
454  keys.Add(new NStorageKey(key));
455  }
456  pair.Key(new NResultSet<INStorageKey>(keys, null));
457  break;
458  case Envelope.PayloadOneofCase.StorageData:
459  var storageData = new List<INStorageData>();
460  foreach (var data in message.StorageData.Data)
461  {
462  storageData.Add(new NStorageData(data));
463  }
464  pair.Key(new NResultSet<INStorageData>(storageData, null));
465  break;
466  case Envelope.PayloadOneofCase.Topics:
467  var topics = new List<INTopic>();
468  foreach (var topic in message.Topics.Topics)
469  {
470  topics.Add(new NTopic(topic));
471  }
472  pair.Key(new NResultSet<INTopic>(topics, null));
473  break;
474  case Envelope.PayloadOneofCase.TopicMessageAck:
475  pair.Key(new NTopicMessageAck(message.TopicMessageAck));
476  break;
477  case Envelope.PayloadOneofCase.TopicMessages:
478  var topicMessages = new List<INTopicMessage>();
479  foreach (var topicMessage in message.TopicMessages.Messages)
480  {
481  topicMessages.Add(new NTopicMessage(topicMessage));
482  }
483  pair.Key(new NResultSet<INTopicMessage>(topicMessages,
484  new NCursor(message.TopicMessages.Cursor)));
485  break;
486  case Envelope.PayloadOneofCase.Users:
487  var users = new List<INUser>();
488  foreach (var user in message.Users.Users)
489  {
490  users.Add(new NUser(user));
491  }
492  pair.Key(new NResultSet<INUser>(users, null));
493  break;
494  case Envelope.PayloadOneofCase.Leaderboards:
495  var leaderboards = new List<INLeaderboard>();
496  foreach (var leaderboard in message.Leaderboards.Leaderboards)
497  {
498  leaderboards.Add(new NLeaderboard(leaderboard));
499  }
500  pair.Key(new NResultSet<INLeaderboard>(leaderboards, new NCursor(message.Leaderboards.Cursor)));
501  break;
502  case Envelope.PayloadOneofCase.LeaderboardRecords:
503  var leaderboardRecords = new List<INLeaderboardRecord>();
504  foreach (var leaderboardRecord in message.LeaderboardRecords.Records)
505  {
506  leaderboardRecords.Add(new NLeaderboardRecord(leaderboardRecord));
507  }
508  var cursor = message.LeaderboardRecords.Cursor == null ? null : new NCursor(message.LeaderboardRecords.Cursor);
509  pair.Key(new NResultSet<INLeaderboardRecord>(leaderboardRecords, cursor));
510  break;
511  case Envelope.PayloadOneofCase.Rpc:
512  pair.Key(new NRuntimeRpc(message.Rpc));
513  break;
514  case Envelope.PayloadOneofCase.Notifications:
515  var notifications = new List<INNotification>();
516  foreach (var n in message.Notifications.Notifications)
517  {
518  notifications.Add(new NNotification(n));
519  }
520  var resumableCursor = message.Notifications.ResumableCursor == null ? null : new NCursor(message.Notifications.ResumableCursor);
521  pair.Key(new NResultSet<INNotification>(notifications, resumableCursor));
522  break;
523  default:
524  Logger.TraceFormatIf(Trace, "Unrecognized protocol message: {0}", message);
525  break;
526  }
