Detailed Description

A client for Nakama server which will dispatch all actions on the Unity main thread.

Wrap a socket client NClient so all actions can be dispatched on the Unity main thread.

public class NakamaSessionManager : MonoBehaviour {
private INClient _client;
private void Start() {
var client = NClient.Default("somesecret");
_client = new NManagedClient(client);
}
private void Update() {
(_client as NManagedClient).ExecuteActions();
}
}
